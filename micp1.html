1. Write a program for bcd to binary conversion.

def bcd_to_binary(bcd_num): binary_num = "" for digit in bcd_num: binary_num += bin(int(digit, 16))[2:].zfill(4) return binary_num bcd_number = input("Enter a BCD number: ") binary_result = bcd_to_binary(bcd_number) print("Binary equivalent:", binary_result)
